Complétez le programme pour réaliser le jeu du trop-grand trop-petit afin de retrouver le code secret. 


		 
		 
	

	

		

		

	

	
# Voici la réponse
from random import *
#?
code = randint(10,100)
message = "Quel est mon code secret ?"
while True:
    n = int(input(message))
    if n > code:
        message = "Trop grand..."
    elif n < code:
        message = "Trop petit..."
    else:
        print("Gagné c'est", code)
        break

Complétez le programme pour réaliser le jeu du plus ou moins afin de trouver le nombre secret. 


		 
		 
	

	

		

		

	

	
# Voici la réponse
from random import *
#?
secret = randint(10,100)
message = "Quel est mon nombre secret ?"
while True:
    n = int(input(message))
    if secret > n:
        message = "C'est plus..."
    elif secret < n:
        message = "C'est moins..."
    else:
        print("Gagné c'est", secret)
        break

Complétez l'avant dernière ligne du programme pour déchiffrer le texte msgSecret.
Vous utiliserez fonction chiffrerCesar(), qui renvoie un message chiffré selon le Chiffre de César, et dont les deux paramètres sont : la chaîne de caractères à chiffrer et le décalage (compris entre 1 et 25) à appliquer sur chaque lettre de l'alphabet. 


		 
		 
	

	

		

		

	

	
# Voici la solution

def chiffrerCesar(msgClair, decalage = 3):
    ALPHABET = "abcdefghijklmnopqrstuvwxyz" * 2
    ALPHABET += ALPHABET.upper()
    msgCode = ""
    for lettre in msgClair:
        if lettre in ALPHABET:
            i = ALPHABET.index(lettre)
            lettre = ALPHABET[i + decalage]
        msgCode += lettre
    return msgCode

msgSecret = "Jxhuuh hw Sdla"        #? exemple
# reculer de 3 lettres revient à avancer de 23
msgClair = chiffrerCesar(msgSecret, 23)
print(msgClair)
